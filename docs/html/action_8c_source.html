111void fileAction (Command *command, char *message) {
112 // /file -send filename Envoie un fichier au serveur.
113 // /file -get filename Récupère un fichier du serveur.
114 // /file username filename Envoie un fichier à un autre utilisateur connecté.
115
116 // Get regex groups.
117 char *regexGroupList[3];
118 getRegexGroup(regexGroupList, message, command->regex);
119 // regexGroupList[0] = message
120 // regexGroupList[1] = -send || -get || username
121 // regexGroupList[2] = filename
122
123 pthread_t fileThread;
124
125 if (strcmp("-send", regexGroupList[1]) == 0) {
126 // User send file to the server.
127 printf("file -send filename\n");
128 pthread_create(&fileThread, NULL, receiveFileThreaded, regexGroupList[2]);
129 } else if (strcmp("-get", regexGroupList[1]) == 0) {
130 // User get file from the server.
131 printf("file -get filename\n");
132 pthread_create(&fileThread, NULL, sendFileThreaded, regexGroupList[2]);
133 } else {
134 // User send file to other user.
135 printf("file username filename\n");
136
137 struct paramFileThreaded *param;
138 param = malloc(sizeof(struct paramFileThreaded));
139 param->message = message;
140 param->username = regexGroupList[1]; // -send || -get || username
141
142 pthread_create(&fileThread, NULL, mpFileThreaded, param);
143
144 free(regexGroupList[0]);
145// free(regexGroupList[1]);
146// free(regexGroupList[2]);
147 }
148}

298void joinAction (Client *client, Command *command, char *message) {
299 // Check that wanted channel exists.
300 // Send the port of this socket.
301 // Wait for client's connection to the new socket.
302 // Close old socket. Or close from client ?
303 // Modify client->socketDescriptor.
304
305 // Get regex groups.
306 char *regexGroupList[3];
307 getRegexGroup(regexGroupList, message, command->regex);
308 int channelIndex = atoi(regexGroupList[1]);
309
310 // Connect to the switch socket.
311 int clientSwitchSocket = connectToClient(channelList[INDEX_SWITCH_CHANNEL]->serverSocketDescriptor);
312
313 if (channelIndex == client->indexCurrentChannel) {
314 // User already in this channel.
315 // Send error status.
316 sendMessageInt(clientSwitchSocket, 400);
317 }
318 else if (channelIndex < 0 || channelIndex >= NB_CHANNEL || ! channelList[channelIndex]->isPublic) {
319 // Channel invalid.
320 // Send error status.
321 sendMessageInt(clientSwitchSocket, 404);
322 }
323 else if (isFull(channelList[channelIndex])) {
324 // Channel already full.
325 // Send error status.
326 sendMessageInt(clientSwitchSocket, 409);
327 }
328 else {
329 // Valid channel.
330 // Connect to switch channel and send port in it.
331 sendMessageInt(clientSwitchSocket, channelList[channelIndex]->port);
332 close(clientSwitchSocket);
333
334 int newClientSocket = connectToClient(channelList[channelIndex]->serverSocketDescriptor);
335
336 int oldSocket = client->acceptedSocketDescriptor;
337 int oldIndexChannel = client->indexCurrentChannel;
338 client->acceptedSocketDescriptor = newClientSocket;
339 client->indexCurrentChannel = channelIndex;
340
341
342 char *resetMessage = "\0"; // Need to send this reset message because the client is blocked in receiveMessage(...).
343 sendMessageInt(oldSocket, 1); // Client blocked in recv int with the old socket.
344 sendMessageString(newClientSocket, resetMessage, 1);
345
346 // Get old channel name.
347 char oldChannelName[100];
348 getChannelName(channelIndex, oldChannelName, 100);
349 sendMessage(oldSocket, oldChannelName); // Send the new channel name to the user.
350 close(oldSocket);
351
352 delete(channelList[oldIndexChannel]->clientList, client);
353 add(channelList[channelIndex]->clientList, client);
354 }
