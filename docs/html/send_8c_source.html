97void sendFile(int clientSocketForFile, FILE *file){
98 // Get file blocSize.
99 long fileSize;
100 fseek(file, 0, SEEK_END); // Jump to the end of the file.
101 fileSize = ftell(file); // Get the current byte offset in the file.
102 rewind(file); // Jump back to the beginning of the file.
103
104 // Send file blocSize to the server.
105 send(clientSocketForFile, &fileSize, sizeof(long), 0);
106
107 // Send the file bloc per bloc.
108 long blocSize = MAX_SIZE_SENT;
109 char subBuffer[blocSize];
110 for (int i = 0; i < fileSize; i += MAX_SIZE_SENT) {
111 blocSize = (i + MAX_SIZE_SENT < fileSize) ? MAX_SIZE_SENT : fileSize - i; // Calcul the bloc's size.
112 fread(subBuffer, blocSize, 1, file); // Read in the file.
113
114 if (send(clientSocketForFile, subBuffer, sizeof(subBuffer), 0) == -1) { // Send bloc of data.
115 throwError("Error in sending file. \n", 1);
116 }
117 bzero(subBuffer, MAX_SIZE_SENT); // Clear the buffer.
118 }
119
120 fclose(file); // Close the file
121 printf("File sent. \n");
122}
